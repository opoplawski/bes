64 void function_scale_grid(int argc, BaseType *argv[], DDS &, BaseType **btpp)

69  "</function>\n";
70 
71  if (argc == 0) {
72  auto_ptr<Str> response(new Str("info"));
73  response->set_value(info);
74  *btpp = response.release();

76  }
77 
78  if (argc < 3 || argc > 5) {
79  throw Error("The scale_grid() function requires three arguments: a Grid and the new lon, lat extents (got " + long_to_string(argc) + " args).\n\
80  See http://docs.opendap.org/index.php/Server_Side_Processing_Functions#scale_grid");
81  }
82 
83  Grid *src = dynamic_cast < Grid * >(argv[0]);
84  if (!src)

86

88  unsigned long y = extract_uint_value(argv[1]);
89  unsigned long x = extract_uint_value(argv[2]);
90 
91  string crs = "WGS84";
92  string interp = "nearest";
93  if (argc > 3) {
94  crs = extract_string_argument(argv[3]);
95  }

97  interp = extract_string_argument(argv[4]);
98  }
99 
100  BESDEBUG(DEBUG_KEY,"function_scale_grid() - CRS '"<< crs << "'" << endl);
101  BESDEBUG(DEBUG_KEY,"function_scale_grid() - Interpolation Method '"<< interp << "'" << endl);
102 
103  SizeBox size(y, x);
104  *btpp = scale_dap_grid(src, size, crs, interp);
105 }
