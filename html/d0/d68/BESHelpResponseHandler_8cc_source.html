69 void
70 BESHelpResponseHandler::execute( BESDataHandlerInterface &dhi )
71 {
72  BESInfo *info = BESInfoList::TheList()->build_info() ;
73  _response = info ;
74 
75  info->begin_response( HELP_RESPONSE_STR, dhi ) ;
76  dhi.action_name = HELP_RESPONSE_STR ;
77 
78  map<string,string> attrs ;
79  attrs["name"] = PACKAGE_NAME ;
80  attrs["version"] = PACKAGE_VERSION ;
81  info->begin_tag( "module", &attrs ) ;
82  info->add_data_from_file( "BES.Help", "BES Help" ) ;
83  info->end_tag( "module" ) ;
84 
85  // execute help for each registered request server
86  info->add_break( 2 ) ;
87 
88  BESRequestHandlerList::TheList()->execute_all( dhi ) ;
89 
90  info->end_response() ;
91 }
