63 bool
64 BESDapSequenceAggregationServer::test_dds_for_suitability(DDS *dds, bool names_must_match)
65 {
66  // DDS *dds = response->get_dds();
67  Sequence *proto; // Will hold the first sequence; use to test uniformity
68  for (DDS::Vars_iter i = dds->var_begin(), e = dds->var_end(); i != e; ++i) {
69  if ((*i)->type() != dods_structure_c)
70  return false;
71  Structure *container = static_cast<Structure*>(*i);
72  if (container->element_count() != 1)
73  return false;
74  if (container->get_var_index(0)->type() != dods_sequence_c)
75  return false;
76 
77  // Now test for uniform Sequences
78  Sequence *s = static_cast<Sequence*>(container->get_var_index(0));
79  if (i == dds->var_begin()) { // The first sequence is the prototype
80  proto = s;
81  for (Sequence::Vars_iter pi = proto->var_begin(), pe = proto->var_end(); pi != pe; ++pi) {
82  if (!(*pi)->is_simple_type())
83  return false;
84  }
85  }
86  else {
87  if (proto->element_count() != s->element_count())
88  return false;
89 
90  Sequence::Vars_iter si = s->var_begin(), se = s->var_end();
91  for (Sequence::Vars_iter pi = proto->var_begin(), pe = proto->var_end(); pi != pe && si != se; ++pi, ++si) {
92  if ((*pi)->type() != (*si)->type())
93  return false;
94  if (names_must_match && ((*pi)->name() != (*si)->name()))
95  return false;
96  }
97  }
98  }
99 
100  return true;
101 }

182 auto_ptr<DataDDS>
183 BESDapSequenceAggregationServer::build_new_dds(DataDDS *source)
184 {
185  auto_ptr<DataDDS> dest(new DataDDS(0));
186 
187  Sequence *proto = 0;
188 
189  for (DDS::Vars_iter i = source->var_begin(), e = source->var_end(); i != e; ++i) {
190  Structure *container = static_cast<Structure*>(*i);
191 
192  Sequence *s = static_cast<Sequence*>(container->get_var_index(0));
193  if (i == source->var_begin()) {
194  // Because this line copies the values of the first sequence,
195  // a minor optimization is to append the subsequent sequence's
196  // values to it. Note that using ptr_duplicate() means that
197  // if 's' points to a specialization of Sequence, we get that
198  // and not a plain instance of Sequence. That's important because
199  // TabularSequence, in particular, and TestSequence, each define
200  // specializations of the code that serializes values to recognize
201  // when a Sequence instance _already_ holds data values.
202  proto = static_cast<Sequence*>(s->ptr_duplicate()); // copies the SequenceValues object
203  }
204  else {
205  SequenceValues &source = s->value_ref(); // vector<BaseTypeRow *>
206  SequenceValues &dest = proto->value_ref();
207 
208  // Copy the BaseType objects used to hold values.
209  for (vector<BaseTypeRow *>::iterator rows_i = source.begin(); rows_i != source.end(); ++rows_i) {
210  // Get the current BaseType Row
211  BaseTypeRow *src_bt_row_ptr = *rows_i;
212  // Create a new row.
213  BaseTypeRow *dest_bt_row_ptr = new BaseTypeRow;
214  // Copy the BaseType objects from a row to new BaseType objects.
215  // Push new BaseType objects onto new row.
216  for (BaseTypeRow::iterator bt_row_iter = src_bt_row_ptr->begin(); bt_row_iter != src_bt_row_ptr->end();
217  ++bt_row_iter) {
218  BaseType *src_bt_ptr = *bt_row_iter;
219  BaseType *dest_bt_ptr = src_bt_ptr->ptr_duplicate();
220  dest_bt_row_ptr->push_back(dest_bt_ptr);
221  }
222  // Push new row onto d_values.
223  dest.push_back(dest_bt_row_ptr);
224  }
225  }
226  }
227 
228  dest->add_var_nocopy(proto);
229 
230  return dest;
231 }

291 
292  BESDEBUG("sa2", "DDS (before aggregation): " << endl);
293  if (BESDebug::IsSet("sa2")) bes_data_dds->get_dds()->print_constrained(*(BESDebug::GetStrm()));
294 
295  auto_ptr<DataDDS> agg_data_dds = build_new_dds(bes_data_dds->get_dds());
296  agg_data_dds->filename("none");

306 
307  bes_data_dds->set_dds(agg_data_dds.get());
308  agg_data_dds.release();
309 
310  dhi.data[POST_CONSTRAINT] = ""; // FIXME Hack
