53 void BESXMLCatalogCommand::parse_request(xmlNode *node)
54 {
55  string name;
56  string value;
57  map<string, string> props;
58  BESXMLUtils::GetNodeInfo(node, name, value, props);
59  if (name != CATALOG_RESPONSE_STR && name != SHOW_INFO_RESPONSE_STR) {
60  string err = "The specified command " + name + " is not a show catalog or show info command";
61  throw BESSyntaxUserError(err, __FILE__, __LINE__);
62  }
63 
64  // the action is the same for show catalog and show info
65  _dhi.action = CATALOG_RESPONSE;
66 
67  // the CATALOG_OR_INFO data value will say if it's a show catalog or
68  // show info
69  if (name == CATALOG_RESPONSE_STR) {
70  _dhi.data[CATALOG_OR_INFO] = CATALOG_RESPONSE;
71  _str_cmd = "show catalog";
72  }
73  else {
74  _dhi.data[CATALOG_OR_INFO] = SHOW_INFO_RESPONSE;
75  _str_cmd = "show info";
76  }
77 
78  // node is an optional property, so could be empty string
79  _dhi.data[CONTAINER] = props["node"];
80  if (!_dhi.data[CONTAINER].empty()) {
81  _str_cmd += " for " + _dhi.data[CONTAINER];
82  }
83  _str_cmd += ";";
84 
85  // now that we've set the action, go get the response handler for the
86  // action
87  BESXMLCommand::set_response();
88 }
