55 void BESUncompress3GZ::uncompress(const string &src, int dest_fd)
56 {
57  // buffer to hold the uncompressed data
58  char in[CHUNK];
59 
60  // open the file to be read by gzopen. If the file is not compressed
61  // using gzip then all this function will do is trasnfer the data to the
62  // destination file.
63  gzFile gsrc = gzopen(src.c_str(), "rb");
64  if (gsrc == NULL) {
65  string err = "Could not open the compressed file " + src;
66  throw BESInternalError(err, __FILE__, __LINE__);
67  }
68 
69  // gzread will read the data in uncompressed. All we have to do is write
70  // it to the destination file.
71  bool done = false;
72  while (!done) {
73  int bytes_read = gzread(gsrc, in, CHUNK);
74  if (bytes_read == 0) {
75  done = true;
76  }
77  else {
78  int bytes_written = write(dest_fd, in, bytes_read);
79  if (bytes_written < bytes_read) {
80  ostringstream strm;
81  strm << "Error writing uncompressed data for file " << gsrc << ": wrote "
82  << bytes_written << " instead of " << bytes_read;
83  gzclose(gsrc);
84 
85  throw BESInternalError(strm.str(), __FILE__, __LINE__);
86  }
87  }
88  }
89 
90  gzclose(gsrc);
91 }
