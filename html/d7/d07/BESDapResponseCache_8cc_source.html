216 bool BESDapResponseCache::is_valid(const string &cache_file_name, const string &dataset)
217 {
218  // If the cached response is zero bytes in size, it's not valid. This is true
219  // because a DAP data object, even if it has no data still has a metadata part.
220  // jhrg 10/20/15
221 
222  off_t entry_size = 0;
223  time_t entry_time = 0;
224  struct stat buf;
225  if (stat(cache_file_name.c_str(), &buf) == 0) {
226  entry_size = buf.st_size;
227  entry_time = buf.st_mtime;
228  }
229  else {
230  return false;
231  }
232 
233  if (entry_size == 0) return false;
234 
235  time_t dataset_time = entry_time;
236  if (stat(dataset.c_str(), &buf) == 0) {
237  dataset_time = buf.st_mtime;
238  }
239 
240  // Trick: if the d_dataset is not a file, stat() returns error and
241  // the times stay equal and the code uses the cache entry.
242 
243  // TODO Fix this so that the code can get a LMT from the correct handler.
244  if (dataset_time > entry_time) return false;
245 
246  return true;
247 }

260 void BESDapResponseCache::read_data_from_cache(const string &cache_file_name, DDS *fdds)
261 {
262  BESDEBUG("dap_response_cache", __PRETTY_FUNCTION__ << " Opening cache file: " << cache_file_name << endl);
263 
264  ifstream data(cache_file_name.c_str());
265 
266  // Rip off the MIME headers from the response if they are present
267  string mime = get_next_mime_header(data);
268  while (!mime.empty()) {
269  mime = get_next_mime_header(data);
270  }
271 
272  // Parse the DDX; throw an exception on error.
273  DDXParser ddx_parser(fdds->get_factory());
274 #if 1
275  // Read the MPM boundary and then read the subsequent headers
276  string boundary = read_multipart_boundary(data);
277 
278  read_multipart_headers(data, "text/xml", dods_ddx);
279 #endif
280  // Parse the DDX, reading up to and including the next boundary.
281  // Return the CID for the matching data part
282  string data_cid;
283  try {
284  ddx_parser.intern_stream(data, fdds, data_cid, boundary);
285  }
286  catch (Error &e) {
287  BESDEBUG("dap_response_cache", "BESDapResponseCache::read_data_from_cache() - [ERROR] DDX Parser Error: " << e.get_error_message() << endl);
288  throw;
289  }
290 #if 1
291  // Munge the CID into something we can work with
292  data_cid = cid_to_header_value(data_cid);
293 
294  // Read the data part's MPM part headers (boundary was read by
295  // DDXParse::intern)
296  read_multipart_headers(data, "application/octet-stream", dods_data_ddx /* old value? dap4_data */, data_cid);
297 #endif
298  // Now read the data
299 
300  // XDRFileUnMarshaller um(data);
301  XDRStreamUnMarshaller um(data);
302  for (DDS::Vars_iter i = fdds->var_begin(); i != fdds->var_end(); i++) {
303  (*i)->deserialize(um, fdds);
304  }
305 }

311 DDS *
312 BESDapResponseCache::get_cached_data_ddx(const string &cache_file_name, BaseTypeFactory *factory,
313  const string &datasset_name)
314 {
315  BESDEBUG("dap_response_cache", __PRETTY_FUNCTION__ << " Reading cache for " << cache_file_name << endl);
316 
317  DDS *fdds = new DDS(factory);
318 
319  fdds->filename(datasset_name);
320 
321  read_data_from_cache(cache_file_name, fdds);
322 
323  fdds->set_factory(0);
324 
325  // mark everything as read. And 'to send.' That is, make sure that when a response
326  // is retrieved from the cache, all of the variables are marked as 'to be sent.'
327  DDS::Vars_iter i = fdds->var_begin();
328  while (i != fdds->var_end()) {
329  (*i)->set_read_p(true);
330  (*i++)->set_send_p(true);
331  }
332 
333  return fdds;
334 }

336 DDS *
337 BESDapResponseCache::cache_dataset(DDS &dds, const string &constraint, BESDapResponseBuilder *rb,
338  ConstraintEvaluator *eval, string &cache_token)
339 {
340  // These are used for the cached or newly created DDS object
341  BaseTypeFactory factory;
342  DDS *fdds;
343 
344  // Build the response_id. Since the response content is a function of both the dataset AND the constraint,
345  // glue them together to get a unique id for the response.
346  string response_id = dds.filename() + "#" + constraint;
347 
348  // Get the cache filename for this thing.
349  string cache_file_name = get_cache_file_name(response_id, /*mangle*/true);
350 
351  BESDEBUG("dap_response_cache", __PRETTY_FUNCTION__ << " cache_file_name: " << cache_file_name << endl);
352  int fd;
353  try {
354  // If the object in the cache is not valid, remove it. The read_lock will
355  // then fail and the code will drop down to the create_and_lock() call.
356  // is_valid() tests for a non-zero object and for d_dateset newer than
357  // the cached object.
358  if (!is_valid(cache_file_name, dds.filename()))
359  purge_file(cache_file_name);
360 
361  if (get_read_lock(cache_file_name, fd)) {
362  BESDEBUG("dap_response_cache", __PRETTY_FUNCTION__ << " Cache hit (1) for: " << cache_file_name << endl);
363  fdds = get_cached_data_ddx(cache_file_name, &factory, dds.filename());
364  }
365  else if (create_and_lock(cache_file_name, fd)) {
366  // If here, the cache_file_name could not be locked for read access;
367  // try to build it. First make an empty file and get an exclusive lock on it.
368  BESDEBUG("dap_response_cache", __PRETTY_FUNCTION__ << " Caching " << cache_file_name << ", constraint: " << constraint << endl);
369 
370  fdds = new DDS(dds);
371  eval->parse_constraint(constraint, *fdds);
372 
373  // FIXME fix the function eval to allow functions and CEs together
374  if (eval->function_clauses()) {
375  DDS *temp_fdds = eval->eval_function_clauses(*fdds);
376  delete fdds;
377  fdds = temp_fdds;
378  }
379 
380  ofstream data_stream(cache_file_name.c_str());
381  if (!data_stream) {
382  throw BESInternalError("Could not open '" + cache_file_name + "' to write cached response.", __FILE__, __LINE__);
383  }
384 
385 #if 1
386  // FIXME Write a better 'serialize' for caching - can just dump data using the
387  // local word order.
388 
389  string start = "dataddx_cache_start", boundary = "dataddx_cache_boundary";
390 
391  // This is a bit of a hack, but it effectively uses ResponseBuilder to write the
392  // cached object/response without calling the machinery in one of the send_*()
393  // methods. Those methods assume they need to evaluate the BESDapResponseBuilder's
394  // CE, which is not necessary and will alter the values of the send_p property
395  // of the DDS's variables.
396  set_mime_multipart(data_stream, boundary, start, dods_data_ddx, x_plain,
397  last_modified_time(rb->get_dataset_name()));
398 
399  // Setting the version to 3.2 causes send_data_ddx to write the MIME headers that
400  // the cache expects.
401  fdds->set_dap_version("3.2");
402 
403  // Use a ConstraintEvaluator that has not parsed a CE so the code can use
404  // the send method(s)
405  ConstraintEvaluator eval;
406 
407  rb->serialize_dap2_data_ddx(data_stream, *fdds, eval, boundary, start);
408 
409  data_stream << CRLF << "--" << boundary << "--" << CRLF;
410 #endif
411  data_stream.close();
412 
413  // Change the exclusive lock on the new file to a shared lock. This keeps
414  // other processes from purging the new file and ensures that the reading
415  // process can use it.
416  exclusive_to_shared_lock(fd);
417 
418  // Now update the total cache size info and purge if needed. The new file's
419  // name is passed into the purge method because this process cannot detect its
420  // own lock on the file.
421  unsigned long long size = update_cache_info(cache_file_name);
422  if (cache_too_big(size)) update_and_purge(cache_file_name);
423  }
424  // get_read_lock() returns immediately if the file does not exist,
425  // but blocks waiting to get a shared lock if the file does exist.
426  else if (get_read_lock(cache_file_name, fd)) {
427  BESDEBUG("dap_response_cache", __PRETTY_FUNCTION__ << " cache hit (2) for: " << cache_file_name << endl);
428  fdds = get_cached_data_ddx(cache_file_name, &factory, dds.get_dataset_name());
429  }
430  else {
431  throw BESInternalError("Cache error! Unable to acquire DAP Response cache.", __FILE__, __LINE__);
432  }
433  }
434  catch (...) {
435  BESDEBUG("dap_response_cache", __PRETTY_FUNCTION__ << " Caught exception, unlocking cache and re-throw." << endl);
436  // I think this call is not needed. jhrg 10/23/12
437  unlock_cache();
438  throw;
439  }
440 
441  cache_token = cache_file_name; // Set this value-result parameter
442  return fdds;
443 }
