80 BESCatalogUtils::BESCatalogUtils(const string &n) :
81  _name(n), _follow_syms(false)
82 {
83  string key = "BES.Catalog." + n + ".RootDirectory";
84  bool found = false;
85  TheBESKeys::TheKeys()->get_value(key, _root_dir, found);
86  if (!found || _root_dir == "") {
87  string s = key + " not defined in BES configuration file";
88  throw BESSyntaxUserError(s, __FILE__, __LINE__);
89  }
90 
91  // TODO access() or stat() would test for existence faster. jhrg 2.25.18
92  DIR *dip = opendir(_root_dir.c_str());
93  if (dip == NULL) {
94  string serr = "BESCatalogDirectory - root directory " + _root_dir + " does not exist";
95  throw BESNotFoundError(serr, __FILE__, __LINE__);
96  }
97  closedir(dip);
98 
99  found = false;
100  key = (string) "BES.Catalog." + n + ".Exclude";
101  vector<string> vals;
102  TheBESKeys::TheKeys()->get_values(key, vals, found);
103  vector<string>::iterator ei = vals.begin();
104  vector<string>::iterator ee = vals.end();
105  for (; ei != ee; ei++) {
106  string e_str = (*ei);
107  if (!e_str.empty() && e_str != ";") BESUtil::explode(';', e_str, _exclude);
108  }
109 
110  key = (string) "BES.Catalog." + n + ".Include";
111  vals.clear();
112  TheBESKeys::TheKeys()->get_values(key, vals, found);
113  vector<string>::iterator ii = vals.begin();
114  vector<string>::iterator ie = vals.end();
115  for (; ii != ie; ii++) {
116  string i_str = (*ii);
117  if (!i_str.empty() && i_str != ";") BESUtil::explode(';', i_str, _include);
118  }
119 
120  key = "BES.Catalog." + n + ".TypeMatch";
121  list<string> match_list;
122  vals.clear();
123  TheBESKeys::TheKeys()->get_values(key, vals, found);
124  if (!found || vals.size() == 0) {
125  string s = key + " not defined in key file";
126  throw BESInternalError(s, __FILE__, __LINE__);
127  }
128  vector<string>::iterator vi = vals.begin();
129  vector<string>::iterator ve = vals.end();
130  for (; vi != ve; vi++) {
131  BESUtil::explode(';', (*vi), match_list);
132  }
133 
134  list<string>::iterator mli = match_list.begin();
135  list<string>::iterator mle = match_list.end();
136  for (; mli != mle; mli++) {
137  if (!((*mli).empty()) && *(mli) != ";") {
138  list<string> amatch;
139  BESUtil::explode(':', (*mli), amatch);
140  if (amatch.size() != 2) {
141  string s = (string) "Catalog type match malformed, " + "looking for type:regexp;[type:regexp;]";
142  throw BESInternalError(s, __FILE__, __LINE__);
143  }
144  list<string>::iterator ami = amatch.begin();
145  handler_regex newval;
146  newval.handler = (*ami);
147  ami++;
148  newval.regex = (*ami);
149  _match_list.push_back(newval);
150  }
151  }
152 
153  key = (string) "BES.Catalog." + n + ".FollowSymLinks";
154  string s_str;
155  TheBESKeys::TheKeys()->get_value(key, s_str, found);
156  s_str = BESUtil::lowercase(s_str);
157  if (s_str == "yes" || s_str == "on" || s_str == "true") {
158  _follow_syms = true;
159  }
160 }

173 bool BESCatalogUtils::include(const string &inQuestion) const
174 {
175  bool toInclude = false;
176 
177  // First check the file against the include list. If the file should be
178  // included then check the exclude list to see if there are exceptions
179  // to the include list.
180  if (_include.size() == 0) {
181  toInclude = true;
182  }
183  else {
184  list<string>::const_iterator i_iter = _include.begin();
185  list<string>::const_iterator i_end = _include.end();
186  for (; i_iter != i_end; i_iter++) {
187  string reg = *i_iter;
188  if (!reg.empty()) {
189  try {
190  // must match exactly, meaning result is = to length of string
191  // in question
192  BESRegex reg_expr(reg.c_str());
193  if (reg_expr.match(inQuestion.c_str(), inQuestion.length())
194  == static_cast<int>(inQuestion.length())) {
195  toInclude = true;
196  }
197  }
198  catch (BESError &e) {
199  string serr = (string) "Unable to get catalog information, "
200  + "malformed Catalog Include parameter " + "in bes configuration file around " + reg + ": "
201  + e.get_message();
202  throw BESInternalError(serr, __FILE__, __LINE__);
203  }
204  }
205  }
206  }
207 
208  if (toInclude == true) {
209  if (exclude(inQuestion)) {
210  toInclude = false;
211  }
212  }
213 
214  return toInclude;
215 }

224 bool BESCatalogUtils::exclude(const string &inQuestion) const
225 {
226  list<string>::const_iterator e_iter = _exclude.begin();
227  list<string>::const_iterator e_end = _exclude.end();
228  for (; e_iter != e_end; e_iter++) {
229  string reg = *e_iter;
230  if (!reg.empty()) {
231  try {
232  BESRegex reg_expr(reg.c_str());
233  if (reg_expr.match(inQuestion.c_str(), inQuestion.length()) == static_cast<int>(inQuestion.length())) {
234  return true;
235  }
236  }
237  catch (BESError &e) {
238  string serr = (string) "Unable to get catalog information, " + "malformed Catalog Exclude parameter "
239  + "in bes configuration file around " + reg + ": " + e.get_message();
240  throw BESInternalError(serr, __FILE__, __LINE__);
241  }
242  }
243  }
244  return false;
245 }

267 unsigned int BESCatalogUtils::get_entries(DIR *dip, const string &fullnode, const string &use_node,
268  BESCatalogEntry *entry, bool dirs_only)
269 {
270  unsigned int cnt = 0;
271 
272  struct stat cbuf;
273  int statret = stat(fullnode.c_str(), &cbuf);
274  if (statret != 0) {
275  if (errno == ENOENT) { // ENOENT means that the path or part of the path does not exist
276  char *s_err = strerror(errno);
277  throw BESNotFoundError((s_err) ? string(s_err) : string("Node ") + use_node + " does not exist", __FILE__,
278  __LINE__);
279  }
280  // any other error means that access is denied for some reason
281  else {
282  char *s_err = strerror(errno);
283  throw BESNotFoundError((s_err) ? string(s_err) : string("Access denied for node ") + use_node, __FILE__,
284  __LINE__);
285  }
286  }
287 
288  struct dirent *dit;
289  while ((dit = readdir(dip)) != NULL) {
290  string dirEntry = dit->d_name;
291  if (dirEntry == "." || dirEntry == "..") {
292  continue;
293  }
294 
295  string fullPath = fullnode + "/" + dirEntry;
296 
297  // Skip this dir entry if it is a sym link and follow links is false
298  if (follow_sym_links() == false) {
299  struct stat lbuf;
300  (void) lstat(fullPath.c_str(), &lbuf);
301  if (S_ISLNK(lbuf.st_mode))
302  continue;
303  }
304 
305  // look at the mode and determine if this is a
306  // directory or a regular file. If it is not
307  // accessible, the stat fails, is not a directory
308  // or regular file, then simply do not include it.
309  struct stat buf;
310  statret = stat(fullPath.c_str(), &buf);
311  if (statret == 0 && S_ISDIR(buf.st_mode)) {
312  if (exclude(dirEntry) == false) {
313  BESCatalogEntry *curr_entry = new BESCatalogEntry(dirEntry, entry->get_catalog());
314 
315  bes_add_stat_info(curr_entry, buf);
316 
317  entry->add_entry(curr_entry);
318 
319  // we don't go further than this, so we need
320  // to add a blank node here so that we know
321  // it's a node (collection)
322  BESCatalogEntry *blank_entry = new BESCatalogEntry(".blank", entry->get_catalog());
323  curr_entry->add_entry(blank_entry);
324  }
325  }
326  else if (statret == 0 && S_ISREG(buf.st_mode)) {
327  if (!dirs_only && include(dirEntry)) {
328  BESCatalogEntry *curr_entry = new BESCatalogEntry(dirEntry, entry->get_catalog());
329  bes_add_stat_info(curr_entry, buf);
330 
331  list<string> services;
332  // TODO use the d_utils object? jhrg 2.26.18
333  isData(fullPath, _name, services);
334  curr_entry->set_service_list(services);
335 
336  bes_add_stat_info(curr_entry, buf);
337 
338  entry->add_entry(curr_entry);
339  }
340  }
341  } // end of the while loop
342 
343  // TODO this always return zero. FIXME jhrg 2.26.18
344  return cnt;
345 }

347 void BESCatalogUtils::display_entry(BESCatalogEntry *entry, BESInfo *info)
348 {
349  string defcatname = BESCatalogList::TheCatalogList()->default_catalog_name();
350 
351  // start with the external entry
352  map<string, string> props;
353  if (entry->get_catalog() == defcatname) {
354  props["name"] = entry->get_name();
355  }
356  else {
357  string name = entry->get_catalog() + "/";
358  if (entry->get_name() != "/") {
359  name = name + entry->get_name();
360  }
361  props["name"] = name;
362  }
363  props["catalog"] = entry->get_catalog();
364  props["size"] = entry->get_size();
365  props["lastModified"] = entry->get_mod_date() + "T" + entry->get_mod_time();
366  if (entry->is_collection()) {
367  props["node"] = "true";
368  ostringstream strm;
369  strm << entry->get_count();
370  props["count"] = strm.str();
371  }
372  else {
373  props["node"] = "false";
374  }
375  info->begin_tag("dataset", &props);
376 
377  list<string> services = entry->get_service_list();
378  if (services.size()) {
379  list<string>::const_iterator si = services.begin();
380  list<string>::const_iterator se = services.end();
381  for (; si != se; si++) {
382  info->add_tag("serviceRef", (*si));
383  }
384  }
