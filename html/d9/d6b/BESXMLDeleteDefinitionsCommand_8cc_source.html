53 void BESXMLDeleteDefinitionsCommand::parse_request(xmlNode *node)
54 {
55  string name;
56  string value;
57  map<string, string> props;
58  BESXMLUtils::GetNodeInfo(node, name, value, props);
59  if (name != DELETE_DEFINITIONS_STR) {
60  string err = "The specified command " + name + " is not a delete definitions command";
61  throw BESSyntaxUserError(err, __FILE__, __LINE__);
62  }
63 
64  // optional property, defaults to default
65  string storage = props["space"];
66  _dhi.data[STORE_NAME] = storage;
67  if (_dhi.data[STORE_NAME].empty()) {
68  _dhi.data[STORE_NAME] = PERSISTENCE_VOLATILE;
69  storage = PERSISTENCE_VOLATILE;
70  }
71 
72  _dhi.action = DELETE_DEFINITIONS;
73 
74  _str_cmd = (string) "delete definitions from " + storage + ";";
75 
76  // now that we've set the action, go get the response handler for the
77  // action
78  BESXMLCommand::set_response();
79 }
