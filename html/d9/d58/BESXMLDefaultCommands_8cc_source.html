139 
140  BESXMLCommand::del_command( SET_CONTEXT_STR);
141  BESXMLCommand::del_command( SET_CONTEXTS_STR);
142 #if USE_CONTEXTS_RESPONSE_HANDLER
143  BESResponseHandlerList::TheList()->remove_handler(SET_CONTEXTS_ACTION);
144 #else

146 #endif
147  BESXMLCommand::del_command( SETCONTAINER_STR);

149  BESXMLCommand::del_command( DELETE_CONTAINER_STR);
150  BESXMLCommand::del_command( DELETE_CONTAINERS_STR);
151  BESXMLCommand::del_command( DELETE_DEFINITION_STR);
152 
153  BESDEBUG("besxml", "Done Removing default commands:" << endl);
154 
155  return true;
156 }
