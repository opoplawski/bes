182 static string lockStatus(const int fd)
183 {
184  struct flock lock_query;
185 
186  lock_query.l_type = F_WRLCK; /* Test for any lock on any part of a file. */
187  lock_query.l_start = 0;
188  lock_query.l_whence = SEEK_SET;
189  lock_query.l_len = 0;
190  lock_query.l_pid = 0;
191 
192  int ret = fcntl(fd, F_GETLK, &lock_query);
193 
194  stringstream ss;
195  ss << endl;
196  if (ret == -1) {
197  ss << "ERROR! fnctl(" << fd << ",F_GETLK, &lock) returned: " << ret << " errno[" << errno << "]: "
198  << strerror(errno) << endl;
199  }
200  else {
201  ss << "SUCCESS. fnctl(" << fd << ",F_GETLK, &lock) returned: " << ret << endl;
202  }
203 
204  ss << "lock_info.l_len: " << lock_query.l_len << endl;
205  ss << "lock_info.l_pid: " << lock_query.l_pid << endl;
206  ss << "lock_info.l_start: " << lock_query.l_start << endl;
207 
208  string type;
209  switch (lock_query.l_type) {
210  case F_RDLCK:
211  type = "F_RDLCK";
212  break;
213  case F_WRLCK:
214  type = "F_WRLCK";
215  break;
216  case F_UNLCK:
217  type = "F_UNLCK";
218  break;
219 
220  }
221 
222  ss << "lock_info.l_type: " << type << endl;
223  ss << "lock_info.l_whence: " << lock_query.l_whence << endl;
224 
225  return ss.str();
226 }

265 bool BESFileLockingCache::m_check_ctor_params()
266 {
267  // Should this really be a fatal error? What about just not
268  // using the cache in this case or writing out a warning message
269  // to the log. jhrg 10/23/15
270  //
271  // Yes, leave this as a fatal error and handle the case when cache_dir is
272  // empty in code that specializes this class. Those child classes are
273  // all singletons and their get_instance() methods need to return null
274  // when caching is turned off. You cannot do that here without throwing
275  // and we don't want to throw an exception for every call to a child's
276  // get_instance() method just because someone doesn't want to use a cache.
277  // jhrg 9/27/16
278  BESDEBUG("cache", "BESFileLockingCache::" <<__func__ << "() - BEGIN" << endl);
279 
280  if (d_cache_dir.empty()) {
281  BESDEBUG("cache", "BESFileLockingCache::" <<__func__ << "() - " <<
282  "The cache directory was not specified. CACHE IS DISABLED." << endl);
283 
284  disable();
285  return false;
286  }
287 
288  int status = mkdir(d_cache_dir.c_str(), 0775);
289  // If there is an error and it's not that the dir already exists,
290  // throw an exception.
291  if (status == -1 && errno != EEXIST) {
292  string err = "The cache directory " + d_cache_dir + " could not be created: " + strerror(errno);
293  throw BESError(err, BES_SYNTAX_USER_ERROR, __FILE__, __LINE__);
294  }
295 
296  if (d_prefix.empty()) {
297  string err = "The cache file prefix was not specified, must not be empty";
298  throw BESError(err, BES_SYNTAX_USER_ERROR, __FILE__, __LINE__);
299  }
300 
301  // I changed this from '<=' to '<' since the code now uses a cache size
302  // of zero to indicate that the cache will never be purged. The other
303  // size-related methods all still work. Since the field is unsigned,
304  // testing for '< 0' is pointless. Later on in this code the value is capped
305  // at MAX_CACHE_SIZE_IN_MEGABYTES (set in this file), which is 2^44.
306  // jhrg 2.28.18
307 #if 0
308  if (d_max_cache_size_in_bytes < 0) {
309  string err = "The cache size was not specified, must be greater than zero";
310  throw BESError(err, BES_SYNTAX_USER_ERROR, __FILE__, __LINE__);
311  }
312 #endif
313 
314  BESDEBUG("cache",
315  "BESFileLockingCache::" << __func__ << "() -" <<
316  " d_cache_dir: " << d_cache_dir <<
317  " d_prefix: " << d_prefix <<
318  " d_max_cache_size_in_bytes: " << d_max_cache_size_in_bytes << endl);
319 
320  enable();
321  return true;
322 }

333 static bool createLockedFile(string file_name, int &ref_fd)
334 {
335  BESDEBUG("cache2", "createLockedFile() - filename: " << file_name <<endl);
336 
337  int fd;
338  if ((fd = open(file_name.c_str(), O_CREAT | O_EXCL | O_RDWR, 0666)) < 0) {
339  switch (errno) {
340  case EEXIST:
341  return false;
342 
343  default:
344  throw BESInternalError(file_name + ": " + get_errno(), __FILE__, __LINE__);
345  }
346  }
347 
348  struct flock *l = lock(F_WRLCK);
349  // F_SETLKW == set lock, blocking
350  if (fcntl(fd, F_SETLKW, l) == -1) {
351  close(fd);
352  ostringstream oss;
353  oss << "cache process: " << l->l_pid << " triggered a locking error: " << get_errno();
354  throw BESInternalError(oss.str(), __FILE__, __LINE__);
355  }
356 
357  BESDEBUG("cache2", "createLockedFile exit: " << file_name <<endl);
358 
359  // Success
360  ref_fd = fd;
361  return true;
362 }

373 bool BESFileLockingCache::m_initialize_cache_info()
374 {
375  BESDEBUG("cache", "BESFileLockingCache::m_initialize_cache_info() - BEGIN" << endl);
376 
377  // The value set in configuration files, etc., is the size in megabytes. The private
378  // variable holds the size in bytes (converted below).
379  d_max_cache_size_in_bytes = min(d_max_cache_size_in_bytes, MAX_CACHE_SIZE_IN_MEGABYTES);
380  d_max_cache_size_in_bytes *= BYTES_PER_MEG;
381  d_target_size = d_max_cache_size_in_bytes * 0.8;
382 
383  BESDEBUG("cache",
384  "BESFileLockingCache::m_initialize_cache_info() - d_max_cache_size_in_bytes: "
385  << d_max_cache_size_in_bytes << " d_target_size: "<<d_target_size<< endl);
386 
387  bool status = m_check_ctor_params(); // Throws BESError on error; otherwise sets the cache_enabled() property
388  if (status) {
389  d_cache_info = BESUtil::assemblePath(d_cache_dir, d_prefix + ".cache_control", true);
390 
391  BESDEBUG("cache", "BESFileLockingCache::m_initialize_cache_info() - d_cache_info: " << d_cache_info << endl);
392 
393  // See if we can create it. If so, that means it doesn't exist. So make it and
394  // set the cache initial size to zero.
395  if (createLockedFile(d_cache_info, d_cache_info_fd)) {
396  // initialize the cache size to zero
397  unsigned long long size = 0;
398  if (write(d_cache_info_fd, &size, sizeof(unsigned long long)) != sizeof(unsigned long long))
399  throw BESInternalError("Could not write size info to the cache info file `" + d_cache_info + "`",
400  __FILE__,
401  __LINE__);
402 
403  // This leaves the d_cache_info_fd file descriptor open
404  unlock_cache();
405  }
406  else {
407  if ((d_cache_info_fd = open(d_cache_info.c_str(), O_RDWR)) == -1) {
408  throw BESInternalError(get_errno(), __FILE__, __LINE__);
409  }
410  }
411 
412  BESDEBUG("cache",
413  "BESFileLockingCache::m_initialize_cache_info() - d_cache_info_fd: " << d_cache_info_fd << endl);
414  }
415 
416  BESDEBUG("cache",
417  "BESFileLockingCache::m_initialize_cache_info() - END [" << "CACHE IS " << (cache_enabled()?"ENABLED]":"DISABLED]") << endl);
418 
419  return status;
420 }

531 bool BESFileLockingCache::get_read_lock(const string &target, int &fd)
532 {
533  lock_cache_read();
534 
535  bool status = true; // Used to support to techniques. jhrg 4/26/17
536 
537 #if USE_GET_SHARED_LOCK
538  status = getSharedLock(target, fd);
539 
540  if (status) m_record_descriptor(target, fd);
541 #else
542  fd = m_find_descriptor(target);
543  // fd == -1 --> The file is not currently open
544  if ((fd == -1) && (fd = open(target.c_str(), O_RDONLY)) < 0) {
545  switch (errno) {
546  case ENOENT:
547  return false; // The file does not exist
548 
549  default:
550  throw BESInternalError(get_errno(), __FILE__, __LINE__);
551  }
552  }
553 
554  // The file might be open for writing, so setting a read lock is
555  // not possible.
556  struct flock *l = lock(F_RDLCK);
557  if (fcntl(fd, F_SETLKW, l) == -1) {
558  return false; // cannot get the lock
559  }
560 
561  m_record_descriptor(target, fd);
562 #endif
563 
564  unlock_cache();
565 
566  return status;
567 }

724 unsigned long long BESFileLockingCache::update_cache_info(const string &target)
725 {
726  unsigned long long current_size;
727  try {
728  lock_cache_write();
729 
730  if (lseek(d_cache_info_fd, 0, SEEK_SET) == -1)
731  throw BESInternalError("Could not rewind to front of cache info file.", __FILE__, __LINE__);
732 
733  // read the size from the cache info file
734  if (read(d_cache_info_fd, &current_size, sizeof(unsigned long long)) != sizeof(unsigned long long))
735  throw BESInternalError("Could not get read size info from the cache info file!", __FILE__, __LINE__);
736 
737  struct stat buf;
738  int statret = stat(target.c_str(), &buf);
739  if (statret == 0)
740  current_size += buf.st_size;
741  else
742  throw BESInternalError("Could not read the size of the new file: " + target + " : " + get_errno(), __FILE__,
743  __LINE__);
744 
745  BESDEBUG("cache", "BESFileLockingCache::update_cache_info() - cache size updated to: " << current_size << endl);
746 
747  if (lseek(d_cache_info_fd, 0, SEEK_SET) == -1)
748  throw BESInternalError("Could not rewind to front of cache info file.", __FILE__, __LINE__);
749 
750  if (write(d_cache_info_fd, &current_size, sizeof(unsigned long long)) != sizeof(unsigned long long))
751  throw BESInternalError("Could not write size info from the cache info file!", __FILE__, __LINE__);
752 
753  unlock_cache();
754  }
755  catch (...) {
756  unlock_cache();
757  throw;
758  }
759 
760  return current_size;
761 }
