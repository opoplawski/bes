211 Array *Geo2DArray::GetDAPArray()
212 {
213  DBG(cerr << "GetDAPArray() - BEGIN" << endl);
214  DBG(cerr << "GetDAPArray() - maptr_DS: " << maptr_DS.get() << endl);
215  DBG(cerr << "GetDAPArray() - raster band count: " << maptr_DS->GetRasterCount() << endl);
216 
217  // There should be just one band
218  if (maptr_DS->GetRasterCount() != 1)
219  throw Error("In function swath2grid(), expected a single raster band.");
220 
221  // Get the x and y dimensions of the raster band
222  int x = maptr_DS->GetRasterXSize();
223  int y = maptr_DS->GetRasterYSize();
224  GDALRasterBand *rb = maptr_DS->GetRasterBand(1);
225  if (!rb)
226  throw Error("Geo2DArray::GetDAPArray() - In function swath2grid(), could not access the raster data.");
227 
228  // Since the Geo2DArray code works with all data values as doubles,
229  // Assume the raster band has GDAL type GDT_Float64, but test anyway
230  if (GDT_Float64 != rb->GetRasterDataType())
231  throw Error("In function swath2grid(), expected raster data to be of type double.");
232 
233  DBG(cerr << "GetDAPArray() - Destination array will have dimensions: " << x << ", " << y << endl);
234 
235  Array *a = new Array(m_src->name(), new Float64(m_src->name()));
236 
237  // Make the result array have two dimensions
238  Array::Dim_iter i = m_src->dim_begin();
239 
240  a->append_dim(x, m_src->dimension_name(i));
241  ++i;
242 
243  if (i == m_src->dim_end())
244  throw Error("Geo2DArray::GetDAPArray() - In function swath2grid(), expected source array to have two dimensions (2).");
245 
246  a->append_dim(y, m_src->dimension_name(i));
247 
248  // Poke in the data values
249  /* RasterIO ( GDALRWFlag eRWFlag,
250  int nXOff,
251  int nYOff,
252  int nXSize,
253  int nYSize,
254  void * pData,
255  int nBufXSize,
256  int nBufYSize,
257  GDALDataType eBufType,
258  int nPixelSpace,
259  int nLineSpace
260  ) */
261  vector<double> data(x * y);
262  rb->RasterIO(GF_Read, 0, 0, x, y, &data[0], x, y, GDT_Float64, 0, 0);
263 
264  // NB: set_value() copies into new storage
265  a->set_value(data, data.size());
266 
267  // Now poke in some attributes
268  // TODO Make these CF attributes
269  string projection_info = maptr_DS->GetProjectionRef();
270  string gcp_projection_info = maptr_DS->GetGCPProjection();
271 
272  // This sets the instance variable that holds the geotransform coefs. These
273  // are needed by the GetDAPGrid() method.
274  if (CE_None != maptr_DS->GetGeoTransform (m_geo_transform_coef))
275  throw Error("Geo2DArray::GetDAPArray() - In function swath2grid(), could not access the geo transform data.");
276 
277  DBG(cerr << "GetDAPArray() - projection_info: " << projection_info << endl);
278  DBG(cerr << "GetDAPArray() - gcp_projection_info: " << gcp_projection_info << endl);
279  DBG(cerr << "GetDAPArray() - geo_transform coefs: " << double_to_string(m_geo_transform_coef[0]) << endl);
280 
281  AttrTable &attr = a->get_attr_table();
282  attr.append_attr("projection", "String", projection_info);
283  attr.append_attr("gcp_projection", "String", gcp_projection_info);
284  for (unsigned int i = 0; i < sizeof(m_geo_transform_coef); ++i) {
285  attr.append_attr("geo_transform_coefs", "String", double_to_string(m_geo_transform_coef[i]));
286  }
287  DBG(cerr << "GetDAPArray() - END" << endl);
288 
289  return a;
290 }

299 Grid *Geo2DArray::GetDAPGrid()
300 {
301  DBG(cerr << "GetDAPGrid() - BEGIN" << endl);
302 
303  Array *a = GetDAPArray();
304  Array::Dim_iter i = a->dim_begin();
305  int lon_size = a->dimension_size(i);
306  int lat_size = a->dimension_size(++i);
307 
308  Grid *g = new Grid(a->name());
309  g->add_var_nocopy(a, array);
310 
311  // Add maps; assume lon, lat; only two dimensions
312  Array *lon = new Array("longitude", new Float64("longitude"));
313  lon->append_dim(lon_size);
314 
315  vector<double> data(max(lon_size, lat_size)); // (re)use this for both lon and lat
316 
317  // Compute values
318  // u = a*x + b*y
319  // v = c*x + d*y
320  // u,v --> x,y --> lon,lat
321  // The constants a, b, c, d are given by the 1, 2, 4, and 5 entries in the geotransform array.
322 
323  if (m_geo_transform_coef[2] != 0)
324  throw Error("The transformed data's Geographic projection should not be rotated.");
325  for (int j = 0; j < lon_size; ++j) {
326  data[j] = m_geo_transform_coef[1] * j + m_geo_transform_coef[0];
327  }
328 
329  // load (copy) values
330  lon->set_value(&data[0], lon_size);
331  // Set the map
332  g->add_var_nocopy(lon, maps);
333 
334  // Now do the latitude map
335  Array *lat = new Array("latitude", new Float64("latitude"));
336  lat->append_dim(lat_size);
337 
338  if (m_geo_transform_coef[4] != 0)
339  throw Error("The transformed data's Geographic projection should not be rotated.");
340  for (int k = 0; k < lat_size; ++k) {
341  data[k] = m_geo_transform_coef[5] * k + m_geo_transform_coef[3];
342  }
343 
344  lat->set_value(&data[0], lat_size);
345  g->add_var_nocopy(lat, maps);
346 
347  DBG(cerr << "GetDAPGrid() - END" << endl);
348 
349  return g;
350 }

421  else
422  resX = (mdSrcGeoMaxX - mdSrcGeoMinX) / (nXSize - 1);
423

429  mi_RectifiedImageXSize = (int) ((360 + mdSrcGeoMaxX - mdSrcGeoMinX) / res) + 1;
430  else
431  mi_RectifiedImageXSize = (int) ((mdSrcGeoMaxX - mdSrcGeoMinX) / res) + 1;
432 
433  mi_RectifiedImageYSize = (int) fabs((mdSrcGeoMaxY - mdSrcGeoMinY) / res) + 1;
434 
435  DBG(cerr << "SetGeoTransform() - Source image size: " << nXSize << ", " << nYSize << endl);
436  DBG(cerr << "SetGeoTransform() - Rectified image size: " << mi_RectifiedImageXSize << ", " << mi_RectifiedImageYSize << endl);
437 
438  md_Geotransform[0] = mdSrcGeoMinX;
439  md_Geotransform[1] = res;
440  md_Geotransform[2] = 0;
441  md_Geotransform[3] = mdSrcGeoMaxY;
442  md_Geotransform[4] = 0;
443  md_Geotransform[5] = -res;
444  mb_GeoTransformSet = TRUE;
445 
446  DBG(cerr << "SetGeoTransform() - END" << endl);
447 
448  return CE_None;

472  int nLatXSize = m_lat->dimension_size(i, true);

475  int nLonXSize = m_lon->dimension_size(i, true);
476  int nLonYSize = m_lon->dimension_size(i + 1, true);
477 
478  DBG(cerr << "SetGeoBBoxAndGCPs() - nXSize: "<< nXSize << " nLatXSize: " << nLatXSize << endl );
479  DBG(cerr << "SetGeoBBoxAndGCPs() - nYSize: "<< nYSize << " nLatYSize: " << nLatYSize << endl );
480 
481  if (nXSize != nLatXSize || nLatXSize != nLonXSize || nYSize != nLatYSize || nLatYSize != nLonYSize)
482  throw Error("SetGeoBBoxAndGCPs() - The size of latitude/longitude and data field does not match.");
483 
484 #if 0
485  /*
486  * Re-sample Standards:
487  * Height | Width
488  * (0, 500) every other one pixel
489  * [500, 1000) every other two pixels

492  * ... ...
493  */
494

496  int ySpace = 1;

509  m_lon->read();

541  gdalCGP.dfGCPPixel = iPixel;

543  gdalCGP.dfGCPY = y;
