42 char *
43 SocketUtilities::ltoa( long val, char *buf, int base)
44 {
45  ldiv_t r ;
46 
47  if( base > 36 || base < 2 ) // no conversion if wrong base
48  {
49  *buf = '\0' ;
50  return buf ;
51  }
52  if(val < 0 )
53  *buf++ = '-' ;
54  r = ldiv( labs( val ), base ) ;
55 
56  /* output digits of val/base first */
57  if( r.quot > 0 )
58  buf = ltoa( r.quot, buf, base ) ;
59 
60  /* output last digit */
61  *buf++ = "0123456789abcdefghijklmnopqrstuvwxyz"[(int)r.rem] ;
62  *buf = '\0' ;
63  return buf ;
64 }
