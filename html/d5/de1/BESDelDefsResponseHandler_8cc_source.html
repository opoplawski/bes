74 void
75 BESDelDefsResponseHandler::execute( BESDataHandlerInterface &dhi )
76 {
77  dhi.action_name = DELETE_DEFINITIONS_STR ;
78  BESInfo *info = new BESSilentInfo() ;
79  d_response_object = info ;
80 
81  string store_name = dhi.data[STORE_NAME] ;
82  if( store_name == "" )
83  store_name = PERSISTENCE_VOLATILE ;
84  BESDefinitionStorage *store =
85  BESDefinitionStorageList::TheList()->find_persistence( store_name ) ;
86  if( store )
87  {
88  bool deleted = store->del_definitions() ;
89  if( !deleted )
90  {
91  string line = (string)"Unable to delete all definitions "
92  + "from definition store \"" + store_name + "\"" ;
93  throw BESSyntaxUserError( line, __FILE__, __LINE__ ) ;
94  }
95  }
96  else
97  {
98  string line = (string)"Definition store \""
99  + store_name
100  + "\" does not exist. Unable to delete." ;
101  throw BESSyntaxUserError( line, __FILE__, __LINE__ ) ;
102  }
103 }
