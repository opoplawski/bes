52 void SetContextsResponseHandler::execute(BESDataHandlerInterface &dhi)
53 {
54  // To record the information in the DHI data[] map, use one value to hold a
55  // list of all of Context names, then use the names to hold the values. Read
56  // this using names = data[CONTEXT_NAMES] and then:
57  //
58  // for each 'name' in names { value = data['context_'name] }
59  //
60  string names = dhi.data[CONTEXT_NAMES];
61  if (names.empty())
62  throw BESSyntaxUserError( "setContexts: No context names found in the data.", __FILE__, __LINE__);
63 
64  BESDEBUG("besxml", "dhi.data[CONTEXT_NAMES]: " << names << endl);
65 
66  istringstream iss(names);
67  string name;
68  iss >> name;
69  while (iss && !name.empty()) {
70 
71  string value = dhi.data[name];
72 
73  if (value.empty())
74  throw BESSyntaxUserError( "setContexts: Context value was not found in the data.", __FILE__, __LINE__);
75 
76  // Remove the CONTEXT_PREFIX before adding the context
77  assert(name.find(CONTEXT_PREFIX) != string::npos); // Is the prefix really there?
78  name.erase(0, sizeof(CONTEXT_PREFIX)-1); // The constant includes the two double quotes
79 
80  BESDEBUG("besxml", "BESContextManager::TheManager()->set_context(" << name << ", " << value << ")" << endl);
81 
82  BESContextManager::TheManager()->set_context(name, value);
83 
84  iss >> name;
85  }
86 
87  // This would be used in the transmit() method below to send a response back to the
88  // BES's client, if this command returned data. Since it does not, this can be NULL
89  // and the transmit() method can be a no-op. jhrg 2/8/18
90  d_response_object = 0;
91 }
